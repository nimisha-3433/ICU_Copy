import js from pyscript import Element, display import pandas as pd import matplotlib.pyplot as plt import io import base64 # Import your exact simulator code from core_simulator import NeuroICU_Engine def run_dashboard_sim(event): # 1. Update UI to show loading state Element("run-btn").element.innerText = "Simulating..." Element("run-btn").element.disabled = True # 2. Get User Inputs from DOM b = int(Element("in_beds").element.value) v = int(Element("in_vents").element.value) i = int(Element("in_icp").element.value) # 3. Run the Simulation (720 hours = 30 days) simulator = NeuroICU_Engine(num_beds=b, num_vents=v, num_icp=i, num_nurses=20) df_patients, df_ops = simulator.run_simulation(duration_hours=720) # 4. Calculate KPIs total_admissions = len(df_patients) deaths = len(df_patients[df_patients['status'] == 'Deceased']) mortality_rate = (deaths / total_admissions * 100) if total_admissions > 0 else 0 # 5. Update DOM text Element("out_admissions").element.innerText = str(total_admissions) Element("out_mortality").element.innerText = f"{mortality_rate:.1f}%" Element("out_cost").element.innerText = f"₹ {simulator.total_cost_inr:,.0f}" # 6. Generate Matplotlib Plot plt.style.use('dark_background') fig, ax = plt.subplots(figsize=(10, 4)) ax.plot(df_ops['hour'], df_ops['bed_occupancy'], label='Bed Occupancy', color='#3b82f6', linewidth=2) ax.plot(df_ops['hour'], df_ops['vent_utilization'], label='Ventilators', color='#ef4444', linewidth=2) # Add capacity lines based on user inputs ax.axhline(y=b, color='#3b82f6', linestyle='--', alpha=0.5, label='Bed Capacity') ax.set_title('30-Day Resource Utilization Tracking') ax.set_xlabel('Simulation Hour') ax.set_ylabel('Active Units') ax.legend(loc='upper right') ax.grid(True, alpha=0.2) fig.tight_layout() # 7. Render Plot to DOM Element("plot-output").element.innerHTML = "" display(fig, target="plot-output") plt.close(fig) # 8. Reset Button Element("run-btn").element.innerText = "Run 30-Day Simulation" Element("run-btn").element.disabled = False
